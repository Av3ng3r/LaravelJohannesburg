NXLog
  • Universal log collector and forwarder
  • Support for various for inputs and outputs
  • Lightweight and efficient
  • Various features (filters, regex, schedules, ...) 
  • Supports multiple platforms such as Windows, UNIX,
    Android device logs, local syslog. etc.
  • Prioritized processing
  • DBI, Kernel, Program, MS EventLog, File, HTTP(S) 
  • TCP, UDP, TLS/SSL, UNIX Sockets, CSV, JSON

    Module pm_buffer

    Type Mem

    Type Disk

<Output log_files>
   Module om_file
   OutputType LineBased
   Truncate FALSE
   # Remove the priority to keep logs clean
   Exec if $raw_event =~ s/(\s+)$// {}
   Exec if $raw_event =~ /^(\<[0-9]*\>)(.*)/ $raw_event = $2;
   Exec if (($SyslogFacility =~ /(AUTH|AUTHPRIV)$/) and \
        ($SyslogSeverity == 'INFO')) { $dst = 'auth.log'; } \
   else if ($SyslogFacility == 'SECURITY') { $dst = 'security'; } \
   else if ($SourceName == 'exim') { drop(); } \
   else if ($SyslogFacility == 'MAIL' and $SyslogSeverity =~ /INFO|NOTICE|WARNING|ERROR|ERR/) { \
        $dst = 'maillog'; } \
   else if ($SyslogFacility == 'CRON') { $dst = 'cron'; } \
   else if ($SyslogSeverity == 'DEBUG') { $dst = 'debug.log'; } \
   else if ((($SyslogFacility == 'CONSOLE') and ($SyslogSeverity == 'INFO')) or \
        ($SyslogSeverity == 'EMERG')) { $dst = 'console.log'; } \
   else $dst = 'messages';
   File '%LOGDIR%' + $dst
</Output>
