function fetch_values($table,$key,$keyfield='key',$orderfield='',$ordersort='',$limit='') {
  $field = "*";
  $key = sql_sanitize($key);
  if (!$orderfield=='') $order = "order by `".sql_sanitize($orderfield)."` $ordersort";
  else $order = '';
  $table = sql_sanitize($table);
  $query = "select $field from $table where `$keyfield`='$key' $order limit $limit";
  if ($result = do_query($query)) {
    return $result;
  } else return fetch_error($query);
}
