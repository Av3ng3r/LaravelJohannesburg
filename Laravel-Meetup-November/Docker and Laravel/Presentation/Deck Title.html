FIG
Single fig.yml file which describes
how all your containers fit together 
"Fast, isolated development environments using Docker". 
laravel-nginx:
  image: laravelmeetup/nginx
  environment:
    - "TZ=Africa/Johannesburg"
  dns:
    - 172.17.42.1
  volumes_from:
    - laravel-data
  links:
    - laravel-phpfpm
  ports:
    - 80:80
    - 443:443
laravel-phpfpm:
  image: laravelmeetup/phpfpm
  environment:
    - "TZ=Africa/Johannesburg"
  dns:
    - 172.17.42.1
  volumes_from:
    - laravel-data
  mem_limit: 256m
  ports:
    - 9000
